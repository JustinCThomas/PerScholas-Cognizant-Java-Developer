-- REQUIREMENT 1

CREATE TABLE STUDENT_INFO (
  STUDENT_ID varchar(10) PRIMARY KEY,
  FIRST_NAME varchar(20),
  LAST_NAME varchar(25),
  ADDRESS varchar(150));
  
CREATE TABLE COURSE_INFO (
  COURSE_CODE varchar(10) PRIMARY KEY,
  COURSE_NAME varchar(20) NOT NULL,
  COURSE_DESCRIPTION varchar(25),
  COURSE_START_DATE Date,
  COURSE_DURATION int,
  NO_OF_PARTICIPANTS int,
  COURSE_TYPE char(3));
  
  
-- REQUIREMENT 2
CREATE TABLE STUDENT_COURSES (
  STUDENT_ID varchar(10) REFERENCES STUDENT_INFO(STUDENT_ID),
  COURSE_CODE varchar(10) REFERENCES COURSE_INFO(COURSE_CODE));
  
  
-- REQUIREMENT 3
CREATE TABLE COURSE_FEES (
  COURSE_CODE varchar(10) CONSTRAINT codefk REFERENCES COURSE_INFO(COURSE_CODE),
  BASE_FEES number(10,0) CONSTRAINT basefeechk check (BASE_FEES > 15000),
  SPECIAL_FEES number(10,0),
  CONSTRAINT basespecialfeechk check (BASE_FEES > SPECIAL_FEES),
  DISCOUNT Number(5,0) CONSTRAINT discountchk check (DISCOUNT >= 5 AND DISCOUNT <= 15));
